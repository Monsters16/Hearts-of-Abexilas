			add_country_leader_role = {
				country_leader = {

				}
				promote_leader = yes
			}
		}
		hidden_effect = {
			random_other_country = {
				limit = {

				}
			set_nationality = {
				target_country = event_target:absolutist_arcael_civil_war
				character = ARC_kvets_prijzleg
			}
		}
	}
